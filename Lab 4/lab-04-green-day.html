raw_totals <-

3. Rescale the variables so that they go from 0-100 instead of the original range. Name the recaled variables *_pomp.

re_green_data <- 
  raw_totals %>% 
  mutate(
    across(
      green_total : open_total, 
      function(x) datawizard::change_scale(x, range(1,5), to = c(0, 100)), 
      .names = "{.col}_pomp"
    ), 
    .after = id
  )
